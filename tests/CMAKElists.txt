cmake_minimum_required(VERSION 3.10)

project(tests)

set(SOURCES
    tests.cpp
    ../src/adding.cpp
)

add_executable(tests ${SOURCES})

target_include_directories(tests PRIVATE 
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/glad/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/glfw/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/googletest/googletest/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/../headers"
)

target_link_libraries(tests PRIVATE 
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/glfw/lib-mingw-w64/libglfw3.a"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/googletest/build/lib/libgtest_main.a"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/googletest/build/lib/libgtest.a"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/googletest/build/lib/libgmock_main.a"
    "${CMAKE_CURRENT_SOURCE_DIR}/../src/googletest/build/lib/libgmock.a"
    "stdc++"
)

set_target_properties(tests PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "../" # output .exe to "./"
)
