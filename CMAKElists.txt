cmake_minimum_required(VERSION 3.10)

project(cpp-opengl)

set(GLFW_BUILD_SHARED_LIBS OFF)
set(CMAKE_BUILD_TYPE Release)
set(GLFW_LIBRARY_TYPE STATIC)
set(GLFW_BUILD_DOCS OFF)
set(GLFW_BUILD_EXAMPLES OFF)
set(GLFW_BUILD_TESTS OFF)
set(GLFW_INSTALL ON)

set(GLFW_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/glfw/build/lib/cmake/glfw3") 
find_package(GLFW3 REQUIRED HINTS ${GLFW_DIR})

message("GLFW Include Directory is: " ${GLFW_INCLUDE_DIRS})
message("GLFW Libraries are: " ${GLFW_LIBRARIES})

set(GLAD_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/glad/include")

target_include_directories(cpp-opengl PRIVATE 
    ${GLFW_INCLUDE_DIRS}
    GLAD_INCLUDE_DIR
)



target_link_libraries(cpp-opengl PRIVATE ${GLFW_LIBRARIES})
